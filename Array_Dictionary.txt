        /* array と dictionary の比較。
         共に３個の配列を作り、その中の一つをprintする場合の違いを検証する */
        var ary = ["アイス","牛乳","卵"]  //arrayは配列の順番に対応したインデックス番号が振られる。
        print(ary[1])  //インデックス番号は「０」から始まるので、指定したい２番目は「１」となる。

        var dic = ["コンビニ":"アイス","薬局":"牛乳","スーパー":"卵"]  
		//dictionaryは番号の代わりにキー(数値でも文字列でもどちらでも)を指定できる。
        print(dic["薬局"]!)  
		//検索のキーとなる文字列を指定する。
		//「！」はつけなくても問題はないが、付けない場合「optional[...]」と表示される。
        
        /* 結果
         arrayには配列の記載順にインデックス番号が付与される。
         dictionaryには配列に対応したキーを設定する。配列の入力順は関係ない。*/
        
        /* それぞれの配列の中から「卵」を削除し、「豆腐」を追加場合の書き方 */
        ary.remove(at: 2)  //「卵」は「３」つ目にあたるので、「２」を指定し削除
        ary.append("豆腐")  //「豆腐」を追加
        
        dic["スーパー"] = nil  //キー「スーパー」に対応する配列を削除
        dic["スーパー"] = "豆腐"  //キー「スーパー」に新たに対応する「豆腐」を追加。
        
        /* for文を使って全ての項目を表示させ、最終的な配列を確認する */
        for task in ary {
            print(task)
        }
        
        for task in dic {
            print(task)
        }
